class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int n = s.length();
        unordered_map<char, int> charMap;
        int maxLength = 0, start = 0;
    
        for (int i = 0; i < n; i++) {
            if (charMap.find(s[i]) != charMap.end()) {
                start = max(start, charMap[s[i]] + 1);
            }
        charMap[s[i]] = i;
        maxLength = max(maxLength, i - start + 1);
        
        }
    
        return maxLength;
    }
};
